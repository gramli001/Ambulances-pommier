function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getActiveSheet(); var action = e.parameter.action; var callback = e.parameter.callback; var result; if (action == "get") { var data = sheet.getDataRange().getValues(); var headers = data[0]; var rows = []; for (var i = 1; i < data.length; i++) { var row = {}; for (var j = 0; j < headers.length; j++) { row[headers[j]] = data[i][j]; } rows.push(row); } result = JSON.stringify(rows); } else if (action == "add") { sheet.appendRow([e.parameter.date, e.parameter.nom, e.parameter.depart, e.parameter.heureDepart, e.parameter.arrivee, e.parameter.heureArrivee, e.parameter.equipage, e.parameter.vehicule, e.parameter.nb]); result = '"OK"'; } else if (action == "delete") { sheet.deleteRow(parseInt(e.parameter.index) + 2); result = '"OK"'; } else { result = '"OK"'; } if (callback) { return ContentService.createTextOutput(callback + "(" + result + ")").setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(result).setMimeType(ContentService.MimeType.JSON); }
